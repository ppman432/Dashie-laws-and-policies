	mountaineers_decision = {
		
		visible = {
			AND = {
				NOT = {
					has_dlc = "No Compromise, No Surrender"
				}
				NOT = {
					has_dlc = "Arms Against Tyranny"
				}
				NOT = {
					has_tech = mountaineers_c
				}
			}
		}
		available = {
			hidden_trigger = { has_army_experience > 50 }
			if = {
				limit = {
					has_country_flag = es_specialist_cooldown
				}
				custom_trigger_tooltip = {
					tooltip = es_specialist_cooldown_days_tt
					NOT = {	has_country_flag = es_specialist_cooldown }
				}
			}
			has_tech = equipment_upgrade_0
			has_tech = tech_support
		}

		custom_cost_trigger = {
			has_army_experience > 50
		}

		custom_cost_text = decision_cost_army_xp_50
		
		days_remove = 75
		
		complete_effect = {
			set_country_flag = { flag = es_specialist_cooldown days = 100 value = 1 }
		}

		remove_effect = {
			if = {
				limit = {
					NOT = {
						has_tech = tech_mountaineers
					}
				}
				set_technology = {
					tech_mountaineers = 1
				}
			}
			else_if = {
				limit = {
					NOT = {
						has_tech = tech_mountaineers2
					}
				}
				set_technology = {
					tech_mountaineers2 = 1
				}
			}
			else_if = {
				limit = {
					NOT = {
						has_tech = tech_mountaineers3
					}
				}
				set_technology = {
					tech_mountaineers3 = 1
				}
			}
			else_if = {
				limit = {
					NOT = {
						has_tech = mountaineers_a
					}
				}
				set_technology = {
					mountaineers_a = 1
				}
			}
			else_if = {
				limit = {
					NOT = {
						has_tech = mountaineers_b
					}
				}
				set_technology = {
					mountaineers_b = 1
				}
			}
			else_if = {
				limit = {
					NOT = {
						has_tech = mountaineers_c
					}
				}
				set_technology = {
					mountaineers_c = 1
				}
			}
		}
